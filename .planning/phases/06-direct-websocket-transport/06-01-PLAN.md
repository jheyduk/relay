    1. **Update telegram-helper.cjs** — add WebSocket-first routing:

    Add new function `sendViaWebSocket(jsonPayload)`:
    ```javascript
    const net = require('net');
    const RELAY_SOCKET = '/tmp/zellij-claude-relay.sock';

    function sendViaWebSocket(jsonPayload) {
      return new Promise((resolve) => {
        const client = net.createConnection(RELAY_SOCKET, () => {
          client.end(JSON.stringify({ action: 'send', payload: jsonPayload }));
        });
        let response = '';
        client.on('data', chunk => response += chunk);
        client.on('end', () => {
          try { resolve(JSON.parse(response)); }
          catch { resolve({ sent: false }); }
        });
        client.on('error', () => resolve({ sent: false }));
        client.setTimeout(2000, () => { client.destroy(); resolve({ sent: false }); });
      });
    }
    ```

    Replace existing `sendRelay(jsonPayload)` with async version:
    ```javascript
    async function sendRelay(jsonPayload) {
      // Try WebSocket first via Unix socket IPC
      const result = await sendViaWebSocket(jsonPayload);
      if (result.sent) return result;

      // Fallback to Telegram only for permission/completion, only if disconnected >30s
      const type = jsonPayload.type;
      const disconnectedFor = result.disconnectedFor || Infinity;
      if ((type === 'permission' || type === 'completion') && disconnectedFor > 30000) {
        return sendViaTelegram(jsonPayload);
      }
      return { sent: false, reason: 'ws_disconnected' };
    }
    ```

    Add `sendViaTelegram(jsonPayload)` — extract existing Telegram send logic:

    function sendViaTelegram(jsonPayload) {
      const relayToken = loadRelayToken();

      return apiCallWithToken(relayToken, 'sendMessage', {
        chat_id: chatId,
        text: JSON.stringify(jsonPayload),
      });
    }

    const RELAY_PIDFILE = '/tmp/zellij-claude-relay.pid';

    function ensureRelayServer() {
      try {
        // Check if already running
        const pid = parseInt(readFileSync(RELAY_PIDFILE, 'utf8').trim());
        process.kill(pid, 0); // Throws if process doesn't exist
        return; // Already running

        // Not running or stale PID — start it

      try {
        const { spawn } = require('child_process');
        const serverPath = require('path').join(__dirname, 'relay-server.cjs');
        const child = spawn('node', [serverPath], {
          detached: true,
          stdio: 'ignore',
        });
        child.unref();
      } catch {}
    }

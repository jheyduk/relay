Add two new functions to relay-server.cjs, placed after the existing `loadOrCreateConfig()` function (around line 65):

1. `loadProjectRootsConfig()` — reads `~/.config/relay/project-roots.json`, returns parsed object. On missing/invalid file, returns default: `{ roots: ["~/prj"], defaultFlags: "--dangerously-skip-permissions", scanDepth: 2 }`. Does NOT write defaults to disk (unlike server.json). Expand `~` to `os.homedir()` in each root path.

```javascript
const PROJECT_ROOTS_CONFIG = path.join(os.homedir(), '.config', 'relay', 'project-roots.json');

function loadProjectRootsConfig() {
  const defaults = {
    roots: ['~/prj'],
    defaultFlags: '--dangerously-skip-permissions',
    scanDepth: 2,
  };
  try {
    const raw = JSON.parse(fs.readFileSync(PROJECT_ROOTS_CONFIG, 'utf8'));
    return {
      roots: Array.isArray(raw.roots) ? raw.roots : defaults.roots,
      defaultFlags: raw.defaultFlags ?? defaults.defaultFlags,
      scanDepth: typeof raw.scanDepth === 'number' ? raw.scanDepth : defaults.scanDepth,
    };
  } catch {
    return defaults;
  }
}
```

2. `scanDirectories(roots, maxDepth)` — for each root, recursively lists directories up to maxDepth levels. Returns sorted array of `{ path, name }` objects. Skips: hidden directories (name starts with `.`), `node_modules`, `__pycache__`. If a root does not exist, skip silently and log warning to stderr. Sort alphabetically by `name` (case-insensitive).

```javascript
function scanDirectories(roots, maxDepth) {
  const SKIP = new Set(['node_modules', '__pycache__']);
  const results = [];

  function scan(dir, depth) {
    if (depth > maxDepth) return;
    let entries;
    try {
      entries = fs.readdirSync(dir, { withFileTypes: true });
    } catch {
      return;
    }
    for (const entry of entries) {
      if (!entry.isDirectory()) continue;
      if (entry.name.startsWith('.')) continue;
      if (SKIP.has(entry.name)) continue;
      const fullPath = path.join(dir, entry.name);
      results.push({ path: fullPath, name: entry.name });
      if (depth < maxDepth) scan(fullPath, depth + 1);
    }
  }

  for (const root of roots) {
    const expanded = root.replace(/^~/, os.homedir());
    if (!fs.existsSync(expanded)) {
      process.stderr.write(`[relay-server] Warning: project root does not exist: ${expanded}\n`);
      continue;
    }
    // Include the root itself if it's a project directory
    scan(expanded, 1);
  }

  results.sort((a, b) => a.name.localeCompare(b.name, undefined, { sensitivity: 'base' }));
  return results;
}
```

} else if (msg.action === 'list_directories') {

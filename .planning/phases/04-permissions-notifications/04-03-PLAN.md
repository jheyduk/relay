1. **ChatViewModel.kt** -- Add two new actions and track per-message sending state:
   - Add `private val _sendingCallbacks = MutableStateFlow<Set<Long>>(emptySet())` to track which message IDs are currently processing a callback
   - Expose in `ChatUiState`: `val sendingCallbackIds: Set<Long> = emptySet()`
   - Update the `combine` to include `_sendingCallbacks`:
     ```kotlin
     val uiState = combine(
         chatRepository.messagesForSession(kuerzel),
         _localState,
         _sendingCallbacks
     ) { messages, local, sendingIds ->
         ChatUiState(messages = messages, kuerzel = kuerzel, isSending = local.isSending,
                     errorMessage = local.errorMessage, sendingCallbackIds = sendingIds)
     }
     ```
   - Add `answerCallback(messageId: Long, response: String)`:
     ```kotlin
     fun answerCallback(messageId: Long, response: String) {
         viewModelScope.launch {
             _sendingCallbacks.update { it + messageId }
             try {
                 chatRepository.answerCallback(messageId, kuerzel, response)
             } catch (e: Exception) {
                 _localState.update { it.copy(errorMessage = "Callback failed: ${e.message}") }
             } finally {
                 _sendingCallbacks.update { it - messageId }
             }
         }
     }
     ```
   - Add `answerQuestion(messageId: Long, option: String)`:
     ```kotlin
     fun answerQuestion(messageId: Long, option: String) {
         viewModelScope.launch {
             _sendingCallbacks.update { it + messageId }
             try {
                 chatRepository.answerCallback(messageId, kuerzel, option)
                 chatRepository.sendMessage(kuerzel, option)
             } catch (e: Exception) {
                 _localState.update { it.copy(errorMessage = "Answer failed: ${e.message}") }
             } finally {
                 _sendingCallbacks.update { it - messageId }
             }
         }
     }
     ```

   ```kotlin
       }
   }
   ```

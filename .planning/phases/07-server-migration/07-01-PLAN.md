**Dispatch mechanism (SERV-03)**: Replace the `msg.action === 'command'` handler. Currently:
```js
execFile('npx', ['zellij-claude', 'send', `@${msg.kuerzel}`, msg.message], ...)
```
Replace with direct Zellij dispatch. The command flow is:
- The app sends `{ action: 'command', kuerzel: 'abc', message: 'hello' }`
- The server must find which Zellij session has a tab named `@abc`
- Then write the message text into that pane followed by Enter

New implementation for `action === 'command'`:
```js
if (msg.action === 'command') {
  const kuerzel = msg.kuerzel;
  const message = msg.message;
  // Read the session ID from the tab file
  const tabFiles = require('fs').readdirSync('/tmp').filter(f => f.startsWith('zellij-claude-tab-'));
  let targetSessionId = null;
  for (const f of tabFiles) {
    try {
      const content = require('fs').readFileSync(`/tmp/${f}`, 'utf8').trim();
      if (content === kuerzel) {
        targetSessionId = f.replace('zellij-claude-tab-', '');
        break;
      }
    } catch {}
  }

  if (!targetSessionId) {
    process.stderr.write(`[relay-server] No session found for @${kuerzel}\n`);
    return;
  }

  // Find the pane ID for this session
  const zellijSession = process.env.ZELLIJ_SESSION_NAME;
  if (!zellijSession) {
    process.stderr.write('[relay-server] ZELLIJ_SESSION_NAME not set\n');
    return;
  }

  // Write chars directly to the focused pane in the target tab
  // First, find the tab and pane using list-panes
  const { execFile } = require('child_process');
  // Escape the message for shell safety and send via write-chars
  // write-chars sends keystrokes to the focused pane
  const chars = message + '\n';
  execFile('zellij', ['--session', zellijSession, 'action', 'write-chars', chars], {
    timeout: 5000,
  }, (err) => {
    if (err) process.stderr.write(`[relay-server] write-chars error: ${err.message}\n`);
  });
}
```

IMPORTANT: The above is the concept. The actual implementation must handle that `zellij action write-chars` writes to the CURRENTLY FOCUSED pane. To target a specific session's pane, we need to:
1. Use `zellij action list-panes --json` to find the pane ID of the tab matching `@{kuerzel}`
2. Use `zellij action focus-tab` to switch to the right tab first, OR use `zellij action write-chars` with the `--pane-id` flag if available

Check the zellij CLI help for `write-chars` to see if `--pane-id` is supported. If not, the approach is:
1. `zellij action go-to-tab-name @{kuerzel}` to focus the tab
2. `zellij action write-chars` to write to the now-focused pane
3. No need to switch back — the user is on mobile anyway

For `/ls`, the server should instead use `zellij action list-tabs --json --state` to list tabs, then format the response and send back to the app. For other commands, evaluate if they're needed (most are Telegram-era relics).

Replace `raw_command` handler with:
```js
else if (msg.action === 'raw_command') {
  if (msg.command === 'ls') {
    // List sessions directly from Zellij
    const { execFile } = require('child_process');
    const zellijSession = process.env.ZELLIJ_SESSION_NAME;
    if (!zellijSession) return;
    execFile('zellij', ['--session', zellijSession, 'action', 'list-tabs', '--json', '--state'], {
      timeout: 5000,
      encoding: 'utf8',
    }, (err, stdout) => {
      if (err) return;
      try {
        const tabs = JSON.parse(stdout);
        const sessions = tabs
          .filter(t => t.name && t.name.startsWith('@'))
          .map(t => ({ name: t.name.slice(1), active: t.active }));
        if (appSocket && appSocket.readyState === 1) {
          appSocket.send(JSON.stringify({
            type: 'session_list',
            sessions,
            timestamp: Date.now(),
          }));
        }
      } catch {}
    });
  }
}
```
